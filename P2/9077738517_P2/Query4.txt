SELECT
    S_NAME, NATIONLIST.CUSTOMER_COUNT
FROM
    (SELECT 
         S_NAME, count (distinct O_CUSTKEY) as CUSTOMER_COUNT
     FROM
         SUPPLIER, CUSTOMER, PARTSUPP, LINEITEM, ORDERS
     WHERE
         S_SUPPKEY = PS_SUPPKEY and
         C_CUSTKEY = O_CUSTKEY and
         PS_PARTKEY = L_PARTKEY and
         PS_SUPPKEY = L_SUPPKEY and
         O_ORDERKEY = L_ORDERKEY
    GROUP BY
         S_NAME) as NATIONLIST
WHERE
    NATIONLIST.CUSTOMER_COUNT > 615
ORDER BY
    NATIONLIST.CUSTOMER_COUNT desc;
